4.4 Smart Contract (Hardhat)
contracts/PersistentCent.sol (ERC‑20 + Presale)

solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract PersistentCent is ERC20, Ownable {
    uint256 public constant RATE = 1000; // 1 ETH = 1000 PCENT
    uint256 public constant CAP = 1000000 * 10**18; // 1M tokens
    uint256 public sold;
    bool public presaleActive = true;

    event TokensPurchased(address buyer, uint256 amount);

    constructor() ERC20("Persistent Cent", "PCENT") Ownable(msg.sender) {
        _mint(address(this), CAP);
    }

    function buyTokens() external payable {
        require(presaleActive, "Presale ended");
        uint256 tokens = msg.value * RATE;
        require(sold + tokens <= CAP, "Cap exceeded");
        sold += tokens;
        _transfer(address(this), msg.sender, tokens);
        emit TokensPurchased(msg.sender, tokens);
    }

    function endPresale() external onlyOwner {
        presaleActive = false;
        // Burn unsold tokens
        uint256 unsold = balanceOf(address(this));
        if (unsold > 0) _burn(address(this), unsold);
    }

    function withdraw() external onlyOwner {
        payable(owner()).transfer(address(this).balance);
    }
}

4.6 API Routes (Next.js)
app/api/subscribe/route.ts (POST)

ts
import { NextResponse } from 'next/server';
import connectDB from '@/lib/mongodb';
import Subscriber from '@/models/Subscriber';

export async function POST(req: Request) {
  try {
    const { email } = await req.json();
    if (!email) return NextResponse.json({ error: 'Email required' }, { status: 400 });
    await connectDB();
    await Subscriber.create({ email });
    return NextResponse.json({ success: true });
  } catch (error: any) {
    if (error.code === 11000) {
      return NextResponse.json({ error: 'Already subscribed' }, { status: 409 });
    }
    return NextResponse.json({ error: 'Server error' }, { status: 500 });
  }
}
app/api/presale-info/route.ts (GET)

ts
import { NextResponse } from 'next/server';
import { ethers } from 'ethers';
import contractABI from '@/lib/contractABI.json';

export async function GET() {
  try {
    const provider = new ethers.JsonRpcProvider(`https://mainnet.infura.io/v3/${process.env.NEXT_PUBLIC_INFURA_ID}`);
    const contract = new ethers.Contract(process.env.NEXT_PUBLIC_CONTRACT_ADDRESS!, contractABI, provider);
    const [sold, cap, rate, active] = await Promise.all([
      contract.sold(),
      contract.CAP(),
      contract.RATE(),
      contract.presaleActive(),
    ]);
    return NextResponse.json({
      sold: sold.toString(),
      cap: cap.toString(),
      rate: rate.toString(),
      active,
    });
  } catch {
    return NextResponse.json({ error: 'Failed to fetch' }, { status: 500 });
  }
}

PresaleWidget.tsx – core interactive component:

tsx
'use client';
import { useState, useEffect } from 'react';
import { useAccount, useBalance, useWriteContract } from 'wagmi';
import { parseEther } from 'ethers';
import contractConfig from '@/lib/contracts';

export default function PresaleWidget() {
  const { address } = useAccount();
  const [amount, setAmount] = useState('');
  const { writeContract, isPending, isSuccess } = useWriteContract();

  const handleBuy = () => {
    if (!amount) return;
    writeContract({
      ...contractConfig,
      functionName: 'buyTokens',
      value: parseEther(amount),
    });
  };

  return (
    <div className="bg-white dark:bg-gray-800 p-8 rounded-2xl shadow-xl max-w-md mx-auto">
      <h3 className="text-2xl font-bold mb-4">Buy Persistent‑Cent</h3>
      <div className="mb-4">
        <p>1 ETH = 1000 PCENT</p>
        <p>Cap: 1,000,000 PCENT</p>
      </div>
      <input
        type="number"
        placeholder="ETH amount"
        value={amount}
        onChange={(e) => setAmount(e.target.value)}
        className="w-full p-3 rounded-lg border mb-4 text-black"
      />
      <button
        onClick={handleBuy}
        disabled={!address || isPending}
        className="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 rounded-lg disabled:opacity-50"
      >
        {isPending ? 'Processing...' : isSuccess ? 'Success!' : 'Buy PCENT'}
      </button>
      {address && <p className="mt-2 text-sm">Connected: {address}</p>}
    </div>
  );
}
